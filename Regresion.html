import numpy as np
import pandas as pd

def forward_selection_regression(df, target, alpha1=1.0, alpha2=1.0):
    features = df.drop(columns=[target]).columns
    corr = df.corr()

    selected = []
    scores = {}

    # Relevancia con target
    relevance = corr[target].abs()

    # Paso 1
    s1 = relevance.drop(target).idxmax()
    selected.append(s1)
    scores[s1] = relevance[s1]

    # Pasos siguientes
    while len(selected) < len(features):
        best_feature = None
        best_score = -np.inf

        for f in features:
            if f in selected:
                continue

            redundancy = np.mean(
                [abs(corr.loc[f, s]) for s in selected]
            )

            score = alpha1 * relevance[f] - alpha2 * redundancy

            if score > best_score:
                best_score = score
                best_feature = f

        selected.append(best_feature)
        scores[best_feature] = best_score

    return selected, scores
